SELECT IL.NAME AS NAME_LIST, I.SUMMARY AS ISSUE
FROM ISSUESLISTS IL JOIN ISSUES I
ON IL.ID = I.ISSUESLIST_ID;

SELECT  U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM USERS U JOIN ISSUES I
ON U.ID = I.USER_ID_ASSIGNESTO
ORDER BY I.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS LICZBA_ZADAN
FROM USERS U JOIN ISSUES I
ON U.ID = I.USER_ID_ASSIGNESTO
GROUP BY I.USER_ID_ASSIGNESTO
ORDER BY I.ID;
